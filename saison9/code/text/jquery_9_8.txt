function getRandomIntInclusive(min, max) {
    min = Math.ceil(min);
    max = Math.floor(max);
    return Math.floor(Math.random() * (max - min +1)) + min;
}
function getRandomFloatInclusive(min, max){
    return (Math.random() * (max - min) + min).toFixed(2);
}
